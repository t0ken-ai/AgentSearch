# AgentSearch runtime dependencies.
#
# This file mirrors the [project] / [project.optional-dependencies] sections
# of pyproject.toml and is provided for environments that prefer
# `pip install -r requirements.txt` (Docker, CI, plain venvs without TOML).
#
# pip install -r requirements.txt  installs everything: core + MCP + ad-library
#                                  raw-HTTP transport + ad-media downloader.
# pip install -e .                 also works and uses pyproject.toml.

# ── Core ───────────────────────────────────────────────────────────────
# Stealth Chromium with patched fingerprint — the entire scraping layer
# of this project is built on top of it. CloakBrowser pulls in Playwright
# (1.60+) automatically.
cloakbrowser>=0.3.31

# Used by `agentsearch extract` for readability + Markdown extraction.
trafilatura>=2.0

# Used by `agentsearch extract` when the target URL serves a PDF
# (Content-Type: application/pdf). Many MMP / ad-intel reports
# (AppsFlyer, Adjust, Branch, businessofapps) ship as PDFs; without
# this, trafilatura would consume binary bytes and return garbage.
# pdfplumber is pure-Python and pulls in pdfminer.six transitively.
pdfplumber>=0.11

# ── Ad-library raw HTTP transport ──────────────────────────────────────
# google_ad_transparency.GoogleAdTransparencyEngine.raw() needs requests
# to bypass the cloakbrowser+stealth+proxy block on Google ATC.
# AdMediaDownloader also uses requests for streaming downloads.
requests>=2.30

# Bundled CA bundle. `agent_search.proxy._ssl_context()` prefers certifi
# over the platform store on Python 3.14 / macOS where the system
# trust isn't always wired into ssl.create_default_context(). It's a
# transitive dep of `requests` already, but pinning it explicitly
# documents the requirement and protects against minimal `requests`
# wheels that might one day drop the bundle.
certifi>=2024.0

# ── MCP (Model Context Protocol) server ────────────────────────────────
# Optional — only needed if you'll run `python -m agent_search.mcp_server`.
# Skip this line in environments that don't host MCP.
mcp>=1.0
